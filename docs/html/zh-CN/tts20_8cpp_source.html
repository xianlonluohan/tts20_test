42 wire_.beginTransmission(i2c_address_);
43 wire_.write(kRxBufferCapacity);
44 EM_CHECK_EQ(wire_.endTransmission(), kI2cEndTransmissionSuccess);
45
46 EM_CHECK_EQ(wire_.requestFrom(i2c_address_, static_cast<uint8_t>(1)), 1);
47
48 while (wire_.available() == 0);
49
50 rx_buffer_capacity_ = wire_.read();

110bool Tts20::Play(const String &text) {
111 EM_CHECK(text.length() > 0);
112
113 const uint8_t header[] = {0xFD, ((text.length() + 2) >> 8) & 0xFF, ((text.length() + 2) & 0xFF), 0x01, 0x04};
114
115 ClearRxBuffer();
116
117 I2cWrite(header, sizeof(header));
118 I2cWrite(reinterpret_cast<const uint8_t *>(text.c_str()), text.length());
119
120 return ReadUntil(kResponseSuccess, kDefaultTimeoutMs);
121}

123bool Tts20::IsBusy() {
124 const auto start_time = millis();
125 while (millis() - start_time < 5000) {
126 ClearRxBuffer();
127
128 const uint8_t command[] = {0xFD, 0x00, 0x01, 0x21};
129 I2cWrite(command, sizeof(command));
130
131 uint8_t data = 0;
132 if (I2cRead(&data, sizeof(data), 500) == sizeof(data)) {
133 if (data == kResponseBusy) {
134 return true;
135 } else if (data == kResponseIdle) {
136 return false;
137 }
138 }
139 }
140
141 return true;
142}

178void Tts20::I2cWrite(const uint8_t *data, const uint16_t size) {
179 EM_CHECK(data != nullptr && size > 0);
180
181 uint16_t offset = 0;
182
183 while (offset < size) {
184 wire_.beginTransmission(i2c_address_);
185 wire_.write(kTxBufferFreeSpace);
186 EM_CHECK_EQ(wire_.endTransmission(), kI2cEndTransmissionSuccess);
187
188 uint8_t segment_length = 0;
189 EM_CHECK_EQ(wire_.requestFrom(i2c_address_, static_cast<uint8_t>(sizeof(segment_length))), sizeof(segment_length));
190
191 while (wire_.available() == 0);
192
193 segment_length = wire_.read();
194
195 if (segment_length == 0) {
196 continue;
197 }
198
199 if (size - offset < segment_length) {
200 segment_length = size - offset;
201 }
202 wire_.beginTransmission(i2c_address_);
203 wire_.write(kTxBuffer);
204 for (uint8_t i = 0; i < segment_length; i++) {
205 wire_.write(data[offset + i]);
206 }
207
208 EM_CHECK_EQ(wire_.endTransmission(), kI2cEndTransmissionSuccess);
209
210 offset += segment_length;
211 }
212
213 while (true) {
214 wire_.beginTransmission(i2c_address_);
215 wire_.write(kTxBufferIsEmpty);
216 EM_CHECK_EQ(wire_.endTransmission(), kI2cEndTransmissionSuccess);
217
218 EM_CHECK_EQ(wire_.requestFrom(i2c_address_, static_cast<uint8_t>(1)), 1);
219
220 while (wire_.available() == 0);
221
222 if (wire_.read() != 0) {
223 break;
224 }
225 }
226}

228uint8_t Tts20::I2cRead(uint8_t *buffer, const uint8_t expected_length, const uint32_t timeout_ms) {
229 EM_CHECK(buffer != nullptr && expected_length > 0);
230
231 uint8_t actual_length = 0;
232 const auto start_time = millis();
233
234 while (actual_length < expected_length && (millis() - start_time) < timeout_ms) {
235 wire_.beginTransmission(i2c_address_);
236 wire_.write(kRxBufferReadPtr);
237 EM_CHECK_EQ(wire_.endTransmission(), kI2cEndTransmissionSuccess);
238
239 uint8_t data[2] = {0};
240 EM_CHECK_EQ(wire_.requestFrom(i2c_address_, static_cast<uint8_t>(sizeof(data))), sizeof(data));
241
242 uint8_t offset = 0;
243 while (offset < sizeof(data)) {
244 if (wire_.available() > 0) {
245 data[offset++] = wire_.read();
246 }
247 }
248
249 if (data[0] == data[1]) {
250 continue;
251 }
252
253 const uint8_t segment_length =
254 min(expected_length - actual_length, min((data[1] + rx_buffer_capacity_ - data[0]) % rx_buffer_capacity_, rx_buffer_capacity_ - data[0]));
255
256 wire_.beginTransmission(i2c_address_);
257 wire_.write(static_cast<uint8_t>(kRxBufferData + data[0]));
258 EM_CHECK_EQ(wire_.endTransmission(), kI2cEndTransmissionSuccess);
259
260 EM_CHECK_EQ(wire_.requestFrom(i2c_address_, segment_length), segment_length);
261
262 const uint16_t target_bytes = actual_length + segment_length;
263 while (actual_length < target_bytes) {
264 if (wire_.available() > 0) {
265 buffer[actual_length++] = wire_.read();
266 }
267 }
268
269 wire_.beginTransmission(i2c_address_);
270 wire_.write(kRxBufferReadPtr);
271 wire_.write(static_cast<uint8_t>((data[0] + segment_length) % rx_buffer_capacity_));
272 EM_CHECK_EQ(wire_.endTransmission(), kI2cEndTransmissionSuccess);
273 }
274 return actual_length;
275}
276bool Tts20::ReadUntil(const uint8_t target_byte, const uint32_t timeout_ms) {
277 uint8_t data = 0;
278 if (I2cRead(&data, sizeof(data), timeout_ms) == sizeof(data) && data == target_byte) {
279 return true;
280 }
281 return false;
282}
